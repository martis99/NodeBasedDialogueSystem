 Node\+Based\+Dialogue\+System is a Unity package designed to facilitate the creation and management of dialogue systems in games. This package leverages Unity\textquotesingle{}s UI Toolkit and provides a flexible, node-\/based editor for designing complex dialogues.

\hypertarget{md__r_e_a_d_m_e_autotoc_md15}{}\doxysection{\texorpdfstring{Features}{Features}}\label{md__r_e_a_d_m_e_autotoc_md15}

\begin{DoxyItemize}
\item Intuitive Node Editor
\item Branching Dialogues
\item Event driven
\item Nodes grouping
\item Save and Load System
\item Undo/redo
\end{DoxyItemize}\hypertarget{md__r_e_a_d_m_e_autotoc_md16}{}\doxysection{\texorpdfstring{Instalation}{Instalation}}\label{md__r_e_a_d_m_e_autotoc_md16}
